def initialize yabu_conf = ''
	@backup = ''
	@log = Log.instance
	if yabu_conf.empty?
		@yabu_config = YabuConfig.new
	else
		@yabu_config = YabuConfig.new yabu_conf
	end
end

def run options={}
	options = {force: false}.merge(options)
	log_the_options options
	find_newest_backup
	# @todo diplay message and exit if there is no backup
	restore options
end
