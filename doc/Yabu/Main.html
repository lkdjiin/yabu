def run
	if ARGV.size == 1 and ARGV[0] == 'recover'
		startToRecover
	elsif ARGV.size == 1 and ARGV[0] == 'backup'
		startToBackup
	elsif ARGV.size == 0
		startToBackup
	else
		puts "Unknown command"
		exit
	end
end
